# 关键字的作用

js中new关键字主要用来实例化构造器来创建 对象。比如：

function Person(name,age,gender){
  this.name = name
  this.age = age
  this.gender = gender
}

const o = new Person("cp",18,"male")//output:{name: "cp", age: 18, gender: "male"}

# 具体实现

function _new(...args) {
  const [constructor, ...otherArgs] = args;
/* 对传进来的参数进行判断，参数必须是一个函数 */
  if(typeof constructor !== 'function') {
      throw new TypeError('constructor is not a function');
  }else if(constructor === Symbol){
    /*ES6 Symbol函数不能作为构造器使用 */
throw new TypeError('constructor is not a function');
   }

  // 1~2 步骤简单写法 const obj = Object.create(constructor.prototype);

  // 1.创建一个空的简单JavaScript对象（即{}）；
  const obj = {};

  // 2.链接该对象（即设置该对象的构造函数）到另一个对象 ；
  Object.setPrototypeOf(obj, constructor.prototype)

  // 3.将新创建的对象作为函数Person 的 this上下文
  const result = constructor.apply(obj, otherArgs);

  // 4.函数返回值做判断，结果是否为对象？，不是返回this。
  return isPrimitive(result) ? obj : result
}

function isPrimitive(value) {
  /* null 由于历史原因 使用 typeof 关键字返回为对象类型 */
  return value == null || ['string', 'number', 'boolean'].includes(typeof(value))
}
